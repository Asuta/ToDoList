提示信息位置测试

测试说明

此页面用于测试提示信息从右上角移动到右下角的修改。点击下面的按钮来测试不同类型的提示信息：

  • 成功提示：显示绿色背景的成功消息
  • 错误提示：显示红色背景的错误消息
  • 警告提示：显示黄色背景的警告消息
  • 信息提示：显示蓝色背景的信息消息
  • 进度提示：显示带进度条的通知

预期结果：所有提示信息都应该从右下角出现，而不是右上角。

测试代码

// 测试成功提示
function showSuccessNotification() {
    showNotification('操作成功完成！', 'success');
}

// 测试错误提示
function showErrorNotification() {
    showNotification('发生了一个错误！', 'error');
}

// 测试警告提示
function showWarningNotification() {
    showNotification('请注意这是一个警告！', 'warning');
}

// 测试信息提示
function showInfoNotification() {
    showNotification('这是一条信息提示', 'info');
}

// 测试进度提示
function testProgressNotification() {
    const notification = showProgressNotification('正在处理...', 0);
    let progress = 0;
    const interval = setInterval(() => {
        progress += 10;
        updateProgressNotification(notification, progress);
        if (progress >= 100) {
            clearInterval(interval);
            setTimeout(() => {
                showNotification('处理完成！', 'success');
            }, 500);
        }
    }, 200);
}

// 测试连续提示
function showMultipleNotifications() {
    showNotification('第一条消息', 'info');
    setTimeout(() => showNotification('第二条消息', 'success'), 1000);
    setTimeout(() => showNotification('第三条消息', 'warning'), 2000);
    setTimeout(() => showNotification('第四条消息', 'error'), 3000);
}

// 测试自动消失
function showAutoHideNotification() {
    showNotification('这条消息2秒后自动消失', 'info', 2000);
}

function showAutoHideNotification5() {
    showNotification('这条消息5秒后自动消失', 'success', 5000);
}

// 测试移动设备提示
function showMobileNotification() {
    showNotification('移动设备测试消息', 'info', 4000);
}
